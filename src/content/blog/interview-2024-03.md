### 程式題：treverse

- **題目**：

```javascript
// 寫一個 treverse function 輸出 11241251
const target = {
  a: { b: 1 },
  c: [1, 2, 4],
  d: null,
  e: [1, 2, { 4: 5 }],
  f: { 4: 1 },
};
```

- **解答**：

```javascript
function traverse(obj) {
  let result = "";

  if (obj !== null) {
    if (typeof obj === "object") {
      for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
          result += traverse(obj[key]);
        }
      }
    } else {
      // 直接加上基本值
      result += obj;
    }
  }

  return result;
}
```
